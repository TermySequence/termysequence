TARGET_LINK_LIBRARIES(mockemulator os common Threads::Threads)

# Formal unit tests
FUNCTION(DEFTEST name)
  ADD_EXECUTABLE(${name} ${name}.cpp)
  ADD_TEST(${name} ${name})
  TARGET_LINK_LIBRARIES(${name} mockemulator l::Cmocka)
ENDFUNCTION()

DEFTEST(simpleupdate)
DEFTEST(fillreplace)
DEFTEST(splitchar)
DEFTEST(removechar)
DEFTEST(mergechars)
DEFTEST(substr)
DEFTEST(remove)
DEFTEST(insert)
DEFTEST(replace)
DEFTEST(erase)
